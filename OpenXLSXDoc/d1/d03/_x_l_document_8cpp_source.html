64 void XLDocument::OpenDocument(const string &fileName)
65 {
66  // Check if a document is already open. If yes, close it.
67  if(m_archive && m_archive->isOpen()) CloseDocument();
68 
69  m_filePath = fileName;
70  m_archive = make_unique<ZipArchive>(m_filePath);
71  m_archive->open(ZipArchive::WRITE);
72 
73  // Open the Relationships and Content_Types files for the document level.
74  m_documentRelationships = make_unique<XLRelationships>(*this, "_rels/.rels");
75  m_contentTypes = make_unique<XLContentTypes>(*this, "[Content_Types].xml");
76 
77  // Create workbook and property items
78  m_docCoreProperties = CreateItem<XLCoreProperties>("docProps/core.xml");
79  m_docAppProperties = CreateItem<XLAppProperties>("docProps/app.xml");
80  m_workbook = CreateItem<XLWorkbook>("xl/workbook.xml");
81 }

126 bool XLDocument::SaveDocumentAs(const string &fileName)
127 {
128  // If the filename is different than the name of the current file, copy the current file to new destination,
129  // close the current zip file and open the new one.
130  if (fileName != m_filePath) {
131  m_archive->discard();
132  std::ifstream src(m_filePath, std::ios::binary);
133  std::ofstream dst(fileName, std::ios::binary);
134  dst << src.rdbuf();
135 
136  m_filePath = fileName;
137  m_archive = make_unique<ZipArchive>(m_filePath);
138  m_archive->open(ZipArchive::WRITE);
139  }
140 
141  // Commit all XML files, i.e. save to the zip file.
142  m_documentRelationships->CommitXMLData();
143  m_contentTypes->CommitXMLData();
144  for (auto file : m_xmlFiles) {
145  file.second->CommitXMLData();
146  }
147 
148  // Close and re-open the zip file, in order to save changes.
149  m_archive->close();
150  m_archive->open(ZipArchive::WRITE);
151  m_xmlData.clear();
152 
153  return true;
154 }
