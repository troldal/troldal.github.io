435  template<typename T, typename std::enable_if<std::is_integral<T>::value, long long int>::type *>
436  void XLCellValue::Set(T numberValue)
437  {
438  if constexpr (std::is_same<T, bool>::value) { // if bool
439  TypeAttribute().set_value("b");
440  ValueNode().text().set(numberValue ? 1 : 0);
441  }
442  else

444  ValueNode().remove_attribute("t");

469  template<typename T, typename std::enable_if<std::is_integral<T>::value, long long int>::type *>
470  T XLCellValue::Get()
471  {
472  if constexpr (std::is_same<T, bool>::value) {
473  if (ValueType() != XLValueType::Boolean) throw XLException("Cell value is not Boolean");
474  return !(std::string_view(ValueNode().text().get()) == "0");
475  } else {
476  if (ValueType() != XLValueType::Integer) throw XLException("Cell value is not Integer");
477  return ValueNode().text().as_llong();
478  }
479  }
480 
487  template<typename T, typename std::enable_if<std::is_floating_point<T>::value, long double>::type *>
488  T XLCellValue::Get()
489  {
490  if (ValueType() != XLValueType::Float) throw XLException("Cell value is not Float");
491  return std::stold(ValueNode().text().get());
492  }

500  template<typename T, typename std::enable_if<std::is_constructible<T, char*>::value && !std::is_same<T, bool>::value, char*>::type *>
501  T XLCellValue::Get()
502  {
503  if (ValueType() != XLValueType::String) throw XLException("Cell value is not String");
504  if (std::string_view(TypeAttribute().value()) == "str") // ordinary string
505  return T(ValueNode().text().get());
506  if (std::string_view(TypeAttribute().value()) == "s") // shared string
507  return T(SharedStringNode(ValueNode().text().as_ullong()).text().get());
508  else throw XLException("Unknown string type");
509  }
