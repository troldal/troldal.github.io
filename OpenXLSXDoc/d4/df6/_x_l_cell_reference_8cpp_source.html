243 std::string XLCellReference::ColumnAsString(unsigned int column)
244 {
245 
246  // If the Column Name has already been stored, look it up.
247  auto iter = s_columnNames.find(column);
248  if (iter != s_columnNames.end()) return iter->second;
249 
250  // Otherwise calculate it
251  string result;
252 
253  if (column <= 1) return std::string("A");
254  else if (column >= 16384) result = std::string("XFD");
255 
256  // If there is one letter in the Column Name:
257  else if (column <= 26) result = s_alphabet.at(column - 1);
258 
259  // If there are two letters in the Column Name:
260  else if (column > 26 && column <= 702) {
261  int lastLetter = (column - 27) % 26;
262  int firstLetter = (column - 27) / 26;
263 
264  result = s_alphabet.at(firstLetter) + s_alphabet.at(lastLetter);
265  }
266 
267  // If there is three letters in the Column Name:
268  else {
269  int lastLetter = (column - 703) % 26;
270  int middleLetter = ((column - 703) / 26) % 26;
271  int firstLetter = (column - 703) / 676;
272 
273  result = s_alphabet.at(firstLetter) + s_alphabet.at(middleLetter) + s_alphabet.at(lastLetter);
274  }
275 
276  // Store the results for later use
277  s_columnNames[column] = result;
278  s_columnNumbers[result] = column;
279  return result;
280 }

285 unsigned int XLCellReference::ColumnAsNumber(const std::string &column)
286 {
287 
288  // If the Column number has already been stored, look it up.
289  auto iter = s_columnNumbers.find(column);
290  if (iter != s_columnNumbers.end()) return iter->second;
291 
292  // Otherwise, calculate it
293  unsigned int length = column.size();
294  unsigned int result = 0;
295 
296  // If the length of the string is only one character, look up the number in the alphabet.
297  if (length == 1) {
298  for (auto letter : s_alphabet) {
299  if (letter.second == column) {
300  result += letter.first + 1;
301  break;
302  }
303  }
304  }
305 
306  // If the string is two characters long...
307  if (length == 2) {
308  for (auto letter : s_alphabet) {

310  result += (letter.first + 1) * 26;
311  break;
312  }
313  }
314 
315  for (auto letter : s_alphabet) {
316  if (letter.second == column.substr(1, 1)) {
317  result += (letter.first + 1);
318  break;
319  }
320  }
321  }
322  // If the string is three characters long...
323  if (length == 3) {
324  for (auto letter : s_alphabet) {

326  result += (letter.first + 1) * 26 * 26;
327  break;
328  }
329  }

355 std::pair<unsigned long, unsigned int> XLCellReference::CoordinatesFromAddress(const std::string &address)
356 {
357  int letterCount = 0;
358 
359  for (auto letter : address) {
360  if (isalpha(letter)) ++letterCount;
361  }
362 
363  int numberCount = address.size() - letterCount;
364 
365  if (letterCount < 1 || letterCount > 3 || numberCount < 1 || numberCount > 7)
366  // If the Address is invalid, return 0,0
367  //return pair<unsigned long, unsigned int>(0, 0);
368  return make_pair(0, 0);
369  else {
370  unsigned int column = ColumnAsNumber(address.substr(0, letterCount));
371  unsigned long row = RowAsNumber(address.substr(letterCount, numberCount));
372 
373  //return pair<unsigned long, unsigned int>(Row, Column);
374  return make_pair(row, column);
375  }
