160 XLCellType XLCellValue::CellType() const
161 {
162  // Determine the type of the Cell
163  // If neither a Type attribute or a value node is present, the cell is empty.
164  if (!m_parentCell.HasTypeAttribute() && !HasValueNode()) {
165  return XLCellType::Empty;
166  }
167 
168  // If a Type attribute is not present, but a value node is, the cell contains a number.
169  else if (!m_parentCell.HasTypeAttribute() && HasValueNode()) {
170  return XLCellType::Number;
171  }
172 
173  // If the cell is of type "s", the cell contains a shared string.
174  else if (m_parentCell.HasTypeAttribute() && string(m_parentCell.TypeAttribute().value()) == "s") {
175  return XLCellType::String;
176  }
177 
178  // If the cell is of type "inlineStr", the cell contains an inline string.
179  else if (m_parentCell.HasTypeAttribute() && string(m_parentCell.TypeAttribute().value()) == "inlineStr") {
180  return XLCellType::String;
181  }
182 
183  // If the cell is of type "str", the cell contains an ordinary string.
184  else if (m_parentCell.HasTypeAttribute() && string(m_parentCell.TypeAttribute().value()) == "str") {
185  return XLCellType::String;
186  }
187 
188  // If the cell is of type "b", the cell contains a boolean.
189  else if (m_parentCell.HasTypeAttribute() && string(m_parentCell.TypeAttribute().value()) == "b") {
190  return XLCellType::Boolean;
191  }
192 
193  // Otherwise, the cell contains an error.
194  else
195  return XLCellType::Error; //the m_typeAttribute has the ValueAsString "e"
196 }
