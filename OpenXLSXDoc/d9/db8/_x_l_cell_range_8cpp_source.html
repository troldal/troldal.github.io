65 XLCellRange &XLCellRange::operator=(const XLCellRange &other)
66 {
67 
68  if (other.NumRows() != NumRows() || other.NumColumns() != NumColumns())
69  throw range_error("Ranges must be of same size and shape!");
70 
71  for (unsigned long r = 1; r <= NumRows(); ++r) {
72  for (unsigned int c = 1; c <= NumColumns(); ++c) {
73  if (other.Cell(r, c) == nullptr) {
74  Cell(r, c)->Value().Clear();
75  }
76  else {

78  }
