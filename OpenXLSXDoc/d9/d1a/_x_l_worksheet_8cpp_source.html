53 bool XLWorksheet::ParseXMLData()
54 {
55 
56  // Get pointers to key nodes in the XML document.
57  InitDimensionNode();
58  InitSheetViewsNode();
59  InitSheetDataNode();
60  InitColumnsNode();
61 
62  // Read the dimensions of the Sheet and set data members accordingly.
63  string dimensions = DimensionNode().attribute("ref").value();
64  SetFirstCell(XLCellReference("A1"));
65  if (dimensions.find(":") == string::npos) SetLastCell(XLCellReference("A1"));
66  else SetLastCell(XLCellReference(dimensions.substr(dimensions.find(":") + 1)));
67 
68  // If Column properties are grouped, divide them into properties for individual Columns.
69  if (m_columnsNode.type() != pugi::node_null) {
70  auto currentNode = ColumnsNode().first_child();
71  while (currentNode != nullptr) {
72  int min = stoi(currentNode.attribute("min").value());
73  int max = stoi(currentNode.attribute("max").value());
74  m_maxColumn = max;
75  if (min != max) {
76  currentNode.attribute("min").set_value(max);
77  for (int i = min; i < max; i++) {
78  auto newnode = ColumnsNode().insert_child_before("col", currentNode);
79  auto attr = currentNode.first_attribute();
80  while (attr) {
81  newnode.append_attribute(attr.name()) = attr.value();
82  attr = attr.next_attribute();
83  }
84  newnode.attribute("min") = i;
85  newnode.attribute("max") = i;
86  }
87  }
88  currentNode = currentNode.next_sibling();
89  }
90  }
91 
92  // Store all Column nodes in the m_columns vector.
93  if (m_columnsNode.type() != pugi::node_null) {
94  auto currentColumn = ColumnsNode().first_child();
95  while (currentColumn) {
96  m_columns.emplace(currentColumn.attribute("min").as_ullong() - 1, XLColumn(*this, currentColumn));
97  currentColumn = currentColumn.next_sibling();
98  }
99  }
100 
101  // Store all Row nodes in the m_rows vector. The XLRow constructor will initialize the cells objects
102  for (auto &currentRow : SheetDataNode().children())
103  m_rows.emplace(stoul(currentRow.attribute("r").value()) - 1, XLRow(*this, currentRow));
104 
105  return true;
106 }

157  unsigned int columnNumber)
158 {
159  // If the requested Cell is outside the current Sheet Range, reset the m_lastCell Property accordingly.

267 XLColumn *XLWorksheet::Column(unsigned int columnNumber)
268 {
269 
270  // If no columns exists, create the <cols> node in the XML document.
271  if (!m_columnsNode) m_columnsNode = XmlDocument()->root().insert_child_before("cols", SheetDataNode());
272 
273  // Create result object and initialize to nullptr.
274  XLColumn *result = nullptr;
275 
276  // Retrieve the Column node in the m_columns vector. If it doesn't exist, nullptr will be returned.
277  XLColumn *colNode = &m_columns.at(columnNumber - 1); // vector is 0-based;
278 
279  // If the node does not exist, create and insert it.
280  if (colNode == nullptr) {
281  // Create the node.
282  XMLNode nodeColumn;
283 
284  // Insert the newly created Column node in the right place in the XML structure.
285  if (!ColumnsNode().first_child() || columnNumber >= m_maxColumn) {
286  // If there are no Column nodes, or the requested Column number exceed the current maximum, append the the node
287  // after the existing Column nodes.
288  nodeColumn = ColumnsNode().append_child("col");
289  }
290  else {
291  //Otherwise, search the Column nodes vector for the next node and insert there.
292  auto index = columnNumber - 1; // vector is 0-based, Excel is 1-based; therefore columnNumber-1.
293  XLColumn *col = &m_columns.at(index);
294  while (col == nullptr) col = &m_columns.at(index++);
295  nodeColumn = ColumnsNode().insert_child_before("col", col->ColumnNode());
296  }
297 
298  nodeColumn.append_attribute("min") = columnNumber;
299  nodeColumn.append_attribute("max") = columnNumber;
300  nodeColumn.append_attribute("width") = 10;
301  nodeColumn.append_attribute("customWidth") = 1;
302 
303  // Insert the new Row node in the Row nodes vector.
304  m_columns.emplace(columnNumber - 1, XLColumn(*this, nodeColumn));
305  result = &m_columns.at(columnNumber - 1);
306  }
307  else {
308  result = colNode;
309  }
310 
311  if (columnNumber > m_maxColumn) m_maxColumn = columnNumber;
312 
313  return result;
314 }

571 void XLWorksheet::Export(const std::string &fileName, char decimal, char delimiter)
572 {
573  ofstream file(fileName);
574  string token;
575  char oldDecimal;
576 
577  if (decimal == ',') oldDecimal = '.';
578  else oldDecimal = ',';
579 
580  for (unsigned long row = 1; row <= RowCount(); ++row) {
581  for (unsigned int column = 1; column <= ColumnCount(); ++column) {
582  token = Cell(row, column)->Value().AsString();
583  replace(token.begin(), token.end(), oldDecimal, decimal);
584  file << token << delimiter;
585  }
586  file << "\n";
587  }
588 
589  file.close();
590 }

595 void XLWorksheet::Import(const std::string &fileName, const string &delimiter)
596 {
597  ifstream file (fileName);
598  string line;
599  unsigned long row = 1;
600  XLTokenizer tokenizer("", delimiter);
601  while (getline(file, line)) {
602  tokenizer.SetString(line);
603  unsigned int column = 1;
604  for (auto &iter : tokenizer.Split()) {
605  if (iter.IsInteger()) Cell(row, column)->Value().Set(iter.AsInteger());
606  if (iter.IsFloat()) Cell(row, column)->Value().Set(iter.AsFloat());
607  if (iter.IsString()) Cell(row, column)->Value().Set(iter.AsString());
608  if (iter.IsBoolean()) {
609  if (iter.AsBoolean() == true) Cell(row, column)->Value().Set(true);
610  else Cell(row, column)->Value().Set(false);
611  }
612  column++;
613  }
614  row++;
615  }
616 
617  file.close();
618 }
