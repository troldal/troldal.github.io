38 bool XLAppProperties::ParseXMLData()
39 {
40  m_sheetNameNodes.clear();
41  m_headingPairs.clear();
42  m_properties.clear();
43 
44  auto node = XmlDocument()->first_child().first_child();
45 
46  while (node) {
47  if (string(node.name()) == "HeadingPairs") {
48  m_headingPairsSize = node.first_child().attribute("size");
49  m_headingPairsCategoryParent = node.first_child().first_child();
50  m_headingPairsCountParent = m_headingPairsCategoryParent.next_sibling();
51 
52  for (int i = 0; i < stoi(m_headingPairsSize.value()) / 2; ++i) {
53  auto categoryNode = m_headingPairsCategoryParent.first_child();
54  auto countNode = m_headingPairsCountParent.first_child();
55 
56  auto element = std::make_pair(categoryNode, countNode);
57 
58  m_headingPairs.push_back(std::move(element));
59 
60  if (i < stoi(m_headingPairsSize.value()) / 2 - 1) {
61  m_headingPairsCategoryParent = m_headingPairsCountParent.next_sibling();
62  m_headingPairsCountParent = m_headingPairsCategoryParent.next_sibling();
63  }
64  }
65  }
66  else if (string(node.name()) == "TitlesOfParts") {
67  m_sheetCountAttribute = node.first_child().attribute("size");
68  m_sheetNamesParent = node.first_child();
69 
70  for (auto &currentNode : m_sheetNamesParent.children())
71  m_sheetNameNodes[currentNode.text().get()] = currentNode;
72  }
73  else {
74  m_properties.insert_or_assign(node.name(), node);
75  }
76 
77  node = node.next_sibling();
78  }
79 
80  return true;
81 }

273 XMLNode XLAppProperties::InsertSheetName(const std::string &sheetName,
274  unsigned int index)
275 {
276  if (index <= 1) return PrependSheetName(sheetName);
277  if (index > m_sheetNameNodes.size()) return AppendSheetName(sheetName);
278 
279  auto curNode = m_sheetNamesParent.first_child();
280  unsigned idx = 1;
281  while (curNode) {
282  if (idx == index) break;
283  curNode = curNode.next_sibling();
284  ++idx;
285  }
286 
287  if (!curNode) return AppendSheetName(sheetName);
288 
289  auto theNode = m_sheetNamesParent.insert_child_before("vt:lpstr", curNode);
290  theNode.text().set(sheetName.c_str());
291 
292  m_sheetNameNodes[sheetName] = theNode;
293  m_sheetCountAttribute.set_value(m_sheetNameNodes.size());
294
