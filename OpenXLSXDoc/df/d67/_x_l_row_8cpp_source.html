20 XLRow::XLRow(XLWorksheet &parent,

42 
43  // Read the hidden attribute
44  auto hiddenAtt = m_rowNode.attribute("hidden");
45  if (hiddenAtt) {

47  else SetHidden(false);
48  }
49 
50  // Iterate throught the Cell nodes and add cells to the m_cells vector
51  auto spansAtt = m_rowNode.attribute("spans");
52  if (spansAtt) {
53  for (auto &currentCell : rowNode.children()) {
54  XLCellReference cellRef(currentCell.attribute("r").value());
55  Resize(cellRef.Column());
56  m_cells.emplace(cellRef.Column() - 1, XLCell::CreateCell(m_parentWorksheet, currentCell));
57  }
58  }

155 XLCell *XLRow::Cell(unsigned int column)
156 {
157  // If the requested Column number is higher than the number of Columns in the current Row,
158  // create a new Cell node, append it to the Row node, resize the m_cells vector, and insert the new node.
159  if (auto result = m_cells.lower_bound(column - 1); result == m_cells.end()) {
160  // Create the new Cell node
161  auto cellNode = m_rowNode.append_child("c");
162  cellNode.append_attribute("r").set_value(XLCellReference(m_rowNumber, column).Address().c_str());
163 
164  // Append the Cell node to the Row node, and create a new XLCell node and insert it in the m_cells vector.
165  m_rowNode.attribute("spans") = string("1:" + to_string(column)).c_str();
166  m_cells.emplace(column - 1, XLCell::CreateCell(m_parentWorksheet, cellNode));
167 
168  // If the requested Column number is lower than the number of Columns in the current Row,
169  // but the Cell does not exist, create a new node and insert it at the rigth position.
170  } else if ((*result).second->CellReference()->Column() != column){
171 
172  // Find the next Cell node and insert the new node at that position.
173  auto cellNode = m_rowNode.insert_child_before("c", (*result).second->CellNode());
174  cellNode.append_attribute("r") = XLCellReference(m_rowNumber, column).Address().c_str();
175  m_cells.emplace(column - 1, XLCell::CreateCell(m_parentWorksheet, cellNode));
176  }
177 
178  return m_cells.at(column - 1).get();
179 }

208 void XLRow::CreateRow(XLWorksheet &worksheet,
209  unsigned long rowNumber)
210 {
211  // Create the node
212  XMLNode nodeRow;
213 
214  // Insert the newly created Row and Cell in the right place in the XML structure.
215  if (!worksheet.SheetDataNode().first_child() || rowNumber >= worksheet.RowCount()) {
216  // If there are no Row nodes, or the requested Row number exceed the current maximum, append the the node
217  // after the existing rownodes.
218  nodeRow = worksheet.SheetDataNode().append_child("row");
219  }
220  else {
221  // Otherwise, search the Row nodes vector for the next node and insert there.
222  // Vector is 0-based, Excel is 1-based; therefore rowNumber-1.
223  XLRows::iterator iter = worksheet.Rows()->lower_bound(rowNumber - 1);
224  if (iter != worksheet.Rows()->end() && iter->second.RowNode().attribute("r").as_ullong() != rowNumber)
225  nodeRow = worksheet.SheetDataNode().insert_child_before("row", iter->second.RowNode());
226  }
227 
228  nodeRow.append_attribute("r") = rowNumber;
229  nodeRow.append_attribute("x14ac:dyDescent") = 0.2;
230  nodeRow.append_attribute("spans") = "1:1";
231  nodeRow.append_child("c").append_attribute("r") = XLCellReference(rowNumber, 1).Address().c_str();
232 
233  worksheet.Rows()->emplace(rowNumber - 1, XLRow(worksheet, nodeRow));
234 }
