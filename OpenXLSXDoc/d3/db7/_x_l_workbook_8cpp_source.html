44 bool XLWorkbook::ParseXMLData()
45 {
46 
47  // Set up the Workbook Relationships.
48  m_relationships.reset(new XLRelationships(*ParentDocument(), "xl/_rels/workbook.xml.rels"));
49 
50  // Find the "sheets" section in the Workbook.xml file
51  for (auto &node : XmlDocument()->first_child().children()) {
52  if (string(node.name()) == "sheets") m_sheetsNode = node;
53  if (string(node.name()) == "definedNames") m_definedNames = node;
54  }
55 
56  for (auto &sheetNode : m_sheetsNode.children())
57  m_sheetNodes[sheetNode.attribute("name").value()] = sheetNode;
58 
59  for (auto const &item : *m_relationships->Relationships()) {
60  string path = item.second->Target();
61 
62  switch (item.second->Type()) {
63 
64  // Set up SharedStrings file
65  case XLRelationshipType::SharedStrings :
66  CreateSharedStrings(*item.second);
67  break;
68 
69  // Set up Styles file
70  case XLRelationshipType::Styles :
71  CreateStyles(*item.second);
72  break;
73 
74  // Set up Worksheet files
75  case XLRelationshipType::Worksheet :
76  CreateWorksheet(*item.second);
77  break;
78 
79  // Set up Chartsheet files
80  case XLRelationshipType::ChartSheet :
81  CreateChartsheet(*item.second);
82  break;
83 
84  default:
85  break;
86  }
87  }
88 
89  return true;
90 }

230 XLRelationshipItem *XLWorkbook::InitiateWorksheet(const std::string &sheetName, unsigned int index)
231 {
232  std::string worksheetPath = "/xl/worksheets/sheet" + to_string(m_sheetId + 1) + ".xml";
233 
234  // Add content item to document
235  ParentDocument()->AddContentItem(worksheetPath, XLContentType::Worksheet);
236 
237  // Add relationship item
238  XLRelationshipItem &item = *m_relationships->AddRelationship(XLRelationshipType::Worksheet,
239  "worksheets/sheet" + to_string(m_sheetId + 1) + ".xml");
240 
241  // insert Sheet node at the given Index
242  if (index == 0) {
243  auto node = m_sheetsNode.append_child("sheet");
244  node.append_attribute("name") = sheetName.c_str();
245  node.append_attribute("sheetId") = to_string(m_sheetId + 1).c_str();
246  node.append_attribute("r:id") = item.Id().c_str();
247  m_sheetNodes[sheetName] = node;
248  }
249  else {
250  auto curNode = m_sheetsNode.first_child();
251  if (!curNode) {
252  auto node = m_sheetsNode.append_child("sheet");
253  node.append_attribute("name") = sheetName.c_str();
254  node.append_attribute("sheetId") = to_string(m_sheetId + 1).c_str();
255  node.append_attribute("r:id") = item.Id().c_str();
256  m_sheetNodes[sheetName] = node;
257  }
258  else {
259  unsigned int idx = 1;
260  while (idx < index) {
261  curNode = curNode.next_sibling();
262  ++idx;
263  if (!curNode) {
264  auto node = m_sheetsNode.append_child("sheet");
265  node.append_attribute("name") = sheetName.c_str();
266  node.append_attribute("sheetId") = to_string(m_sheetId + 1).c_str();
267  node.append_attribute("r:id") = item.Id().c_str();
268  m_sheetNodes[sheetName] = node;
269  }
270  }
271  auto node = m_sheetsNode.insert_child_before("sheet", curNode);
272  node.append_attribute("name") = sheetName.c_str();
273  node.append_attribute("sheetId") = to_string(m_sheetId + 1).c_str();
274  node.append_attribute("r:id") = item.Id().c_str();
275  m_sheetNodes[sheetName] = node;
276  }
277  }
278 
279  // Add entry to the App Properties
280  if (index == 0) ParentDocument()->AppProperties()->InsertSheetName(sheetName, WorksheetCount() + 1);
281  else ParentDocument()->AppProperties()->InsertSheetName(sheetName, index);
282 
283  ParentDocument()->AppProperties()->SetHeadingPair("Worksheets", WorksheetCount() + 1);
284 
285  return &item;
286 }

464 void XLWorkbook::CreateWorksheet(const XLRelationshipItem &item, const std::string& xmlData)
465 {
466 
467  // Find the appropriate sheet node in the Workbook .xml file; get the name and id of the worksheet.
468  string name;
469  for (auto &node : m_sheetsNode.children()) {
470  if (string(node.attribute("r:id").value()) == item.Id()) {
471  name = node.attribute("name").value();
472  break;
473  }
474  }
475 
476  // If xmlData is empty, set the m_sheets and m_childXmlDocuments elements to nullptr. The worksheet will then be
477  // lazy-instantiated when the worksheet is requested using the 'Worksheet" function.
478  // If xmlData is provided (i.e. a new sheet is created or an existing is cloned), create the new worksheets accordingly.
479  m_sheetPaths[name] = "xl/" + item.Target();
480  if (xmlData.empty()) {
481  m_sheets[name] = nullptr;
482  m_childXmlDocuments["xl/" + item.Target()] = nullptr;
483  } else {
484  m_sheets[name] = make_unique<XLWorksheet>(*this, name, "xl/" + item.Target(), xmlData);
485  m_childXmlDocuments["xl/" + item.Target()] = m_sheets.at(name).get();
486  }
487 
488  // Update internal counters.
489  m_sheetCount++;
490  m_worksheetCount++;
491  m_sheetId++;
492 }
